struct person{

struct Node{
struct person data;
struct Node* next;
};
//创建链表
struct Node* createList(){
struct Node* headNode = (struct Node*)malloc(sizeof(struct Node));
headNode->next = NULL;
return headNode;
}

struct Node* createNode(struct person data){
struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
newNode->data = data;
newNode->next = NULL;
return newNode;
}
